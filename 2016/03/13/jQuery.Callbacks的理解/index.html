jQuery.Callbacks的理解

发表于 2016-03-13   |  

一、先看看jQuery.Callbacks()的基本结构

jQuery.Callbacks = function( options ) {
    options = typeof options === "string" ?
        createOptions( options ) :
        jQuery.extend( {}, options );

    var firing,
        memory,
        fired,  
        locked, 
        list = [], 
        queue = [], 
        firingIndex = -1,

    fire = function() {}, //实现回调函数的工具函数
    self = {
        add: function() {}，//添加一个或多个到回调函数列表list中去
        remove: function() {}，//从回调函数列表list中移除一个或多个
        has: function() {}， //判断改函数在回调函数列表中是否存在
        empty: function() {}，//将回调函数列表置为空数组
        disable: function() {}，//将回调函数列表禁用
        disabled: function() {}，//用来判断函数列表是否被禁用
        lock: function() {}，//将回调函数列表锁住
        lockd: function() {}，//判断回调函数列表是否被锁住
        fireWith: function() {}，//使用指定的上下文和参数值，调用回调函数列表的所有函数
        fire: function() {}，//使用指定的参数值，调用回调函数列表的所有函数
        fired: function() {} //判断回调函数列表是否被调用过
    };

    return self; //返回self对象
}

二、其中的具体的源代码

var rnotwhite = ( /\S+/g );

function createOptions( options ) {
    var object = {};
    jQuery.each( options.match( rnotwhite ) || [], function( _, flag ) {
        object[ flag ] = true;
    } );
    return object;
}

jQuery.Callbacks = function( options ) {

    //判断传进来的参数，
    //①如果是字符串的话，调用上面创建对象的函数创建出来，结果如 "once" --> { once : true }
    //②如果不是字符串，则以jQuery.extend 循环遍历，给到options值里面去。
    //总结：其作用即是把 options 解析成对象
    options = typeof options === "string" ?
        createOptions( options ) :
        jQuery.extend( {}, options );

    var 
        firing, // 用来判断回调函数列表是否正在被执行（或者说是否在被使用）
        memory, //作用是记忆之前的一个变量值（保持以前的值）
        fired,  // 判断回调函数list是否被触发过的标记
        locked, // 判断回调函数list是否被锁住的标记。
        list = [], //存放回调函数的列表 
        queue = [], //可重复列表的执行数据队列
        firingIndex = -1, //当前正在执行回调函数列表的下标值

        //该函数作为实际触发回调函数
        fire = function() {
            locked = options.once;
            fired = firing = true;
            for ( ; queue.length; firingIndex = -1 ) {
                memory = queue.shift();
                while ( ++firingIndex < list.length ) {
                    if ( list[ firingIndex ].apply( memory[ 0 ], memory[ 1 ] ) === false &&
                        options.stopOnFalse ) {
                        firingIndex = list.length;
                        memory = false;
                    }
                }
            }

            if ( !options.memory ) {
                memory = false;
            }

            firing = false;
            if ( locked ) {
                if ( memory ) {
                    list = [];
                } else {
                    list = "";
                }
            }
        },

        self = {
            add: function() {
                if ( list ) {
                    if ( memory && !firing ) {
                        firingIndex = list.length - 1;
                        queue.push( memory );
                    }
                    ( function add( args ) {
                        jQuery.each( args, function( _, arg ) {
                            if ( jQuery.isFunction( arg ) ) {
                                if ( !options.unique || !self.has( arg ) ) {
                                    list.push( arg );
                                }
                            } else if ( arg && arg.length && jQuery.type( arg ) !== "string" ) {
                                add( arg );
                            }
                        } );
                    } )( arguments );

                    if ( memory && !firing ) {
                        fire();
                    }
                }
                return this;
            },

            remove: function() {
                jQuery.each( arguments, function( _, arg ) {
                    var index;
                    while ( ( index = jQuery.inArray( arg, list, index ) ) > -1 ) {
                        list.splice( index, 1 );
                        if ( index <= firingIndex ) {
                            firingIndex--;
                        }
                    }
                } );
                return this;
            },

            has: function( fn ) {
                return fn ?
                    jQuery.inArray( fn, list ) > -1 :
                    list.length > 0;
            },

            empty: function() {
                if ( list ) {
                    list = [];
                }
                return this;
            },

            disable: function() {
                locked = queue = [];
                list = memory = "";
                return this;
            },
            disabled: function() {
                return !list;
            },

            lock: function() {
                locked = queue = [];
                if ( !memory ) {
                    list = memory = "";
                }
                return this;
            },
            locked: function() {
                return !!locked;
            },

            fireWith: function( context, args ) {
                if ( !locked ) {
                    args = args || [];
                    args = [ context, args.slice ? args.slice() : args ];
                    queue.push( args );
                    if ( !firing ) {
                        fire();
                    }
                }
                return this;
            },

            fire: function() {
                self.fireWith( this, arguments );
                return this;
            },

            fired: function() {
                return !!fired;
            }
        };

    return self;
};

三、我从中学到了啥。

该jQuery版本2.2.0。之前的版本关于这部分的源码是长的不一样的。
这里的jQuery.Callbacks。采用了设计模式中的观察者模式的发布，订阅机制、

实例：

function a(name){
    console.log("Hello "+name+"!");
}
function b(name){
    console.log("Hi "+name+"!");
}

var cbs = jQuery.Callbacks();//创建回调对象
cbs.add(a);
cbs.fire("World"); 
//输出结果：
//Hello World!

cbs.add(b);
cbs.fire("Earth");
//输出结果：
//Hello Earth!
//Hi Earth!

这就是了。其设计的原理中有个 list 。
其中①创建了一个 jQuery.Callbacks 对象②然后在其里面创建了一个self对象③把self对象返回。这样操作的结果就形成了一个闭包，使得list变量不会被释放。用来存放回调的所有函数。add 函数向list 里面添加。fire 把 list 里面的函数 遍历执行。当改变一个参数值时，list 里面输出的 所有函数值 都会被改变。

即对上了观察者模式的设计场景：当一个对象的改变需要同时改变其它对象，并且它不知道具体有多少对象需要改变的时候，就应该考虑使用观察者模式。
