jQuery源码学习之旅第二感

发表于 2016-02-24   |  

纵观之下，会发现里面大量充斥着，jQuery.extend和jQuery.fn.extend;稍微了解jQuery的人都知道，jQuery.extend()和jQuery.fn.extend()是由jQuery为开发插件提供的两个方法。那么现在来看看这两个方法的源码是怎么样实现的、、、

jQuery.extend = jQuery.fn.extend = function() {
    var options, name, src, copy, copyIsArray, clone,
        target = arguments[ 0 ] || {},
        i = 1,
        length = arguments.length,
        deep = false;

    // 当你传入的第一个值是布尔值，且这个值为true的时候，会进行深拷贝。
    // 同时把第二个值赋给target对象。
    if ( typeof target === "boolean" ) {
        deep = target;
        target = arguments[ i ] || {};
        i++;
    }

    // 这个是当要进行拷贝的target对象，不是对象或者函数。则初始化为空对象
    if ( typeof target !== "object" && !jQuery.isFunction( target ) ) {
        target = {};
    }

    // 当只有一个参数时，则指定jQuery自身作为目标对象
    if ( i === length ) {
        target = this;
        i--;
    }

    for ( ; i < length; i++ ) {

        // 参数值不能为null/undefined
        if ( ( options = arguments[ i ] ) != null ) {

            // Extend the base object
            for ( name in options ) {
                src = target[ name ];
                copy = options[ name ];

                // 阻止无限循环
                if ( target === copy ) {
                    continue;
                }

                //如果对象中包含了数组或者其他对象，则利用递归进行拷贝
                if ( deep && copy && ( jQuery.isPlainObject( copy ) ||
                    ( copyIsArray = jQuery.isArray( copy ) ) ) ) {

                    if ( copyIsArray ) {
                        copyIsArray = false;
                        clone = src && jQuery.isArray( src ) ? src : [];

                    } else {
                        clone = src && jQuery.isPlainObject( src ) ? src : {};
                    }

                    // Never move original objects, clone them
                    target[ name ] = jQuery.extend( deep, clone, copy );

                // 不拷贝undefined值
                } else if ( copy !== undefined ) {
                    target[ name ] = copy;
                }
            }
        }
    }
    // 返回已经被修改的对象
    return target;
};
